async function handleBooking(event) {

event.preventDefault();

 

const name = document.getElementById('name').value;

const phone = document.getElementById('phone').value;

const service = document.getElementById('service').value;

const date = document.getElementById('date').value;

const time = document.getElementById('time').value;

const notes = document.getElementById('notes').value;

const messageBox = document.getElementById('message-box');

 

// 1. 組織預約摘要訊息 (已修改為要求的格式)

const bookingSummary = `【世錦運動按摩預約】

姓名：${name}

聯繫電話：${phone}

預約項目：${service}

時間：${date} ${time}

備註：${notes || '無'}

 

(請勿修改此格式，方便後台快速登記)`;

 

// 2. 嘗試複製到剪貼簿

try {

// 使用 document.execCommand('copy') 作為 fallback

if (!navigator.clipboard) {

// Fallback using execCommand

const textArea = document.createElement("textarea");

textArea.value = bookingSummary;

document.body.appendChild(textArea);

textArea.focus();

textArea.select();

document.execCommand('copy');

document.body.removeChild(textArea);

} else {

await navigator.clipboard.writeText(bookingSummary);

}

 

// 3. 顯示成功訊息並準備跳轉

messageBox.innerHTML = `

<p class="font-bold text-xl" style="color: #06C755;">✅ 預約資訊已複製到剪貼簿！</p>

<p class="text-gray-700 mt-2">即將跳轉至 LINE 聊天室，請在聊天室中**貼上 (Paste)** 訊息即可送出預約。</p>

`;

messageBox.classList.remove('hidden');

 

// 清空表單

document.getElementById('bookingForm').reset();

 

// 延遲 1.5 秒後跳轉

setTimeout(() => {

window.location.href = 'https://lin.ee/6w27dyj';

}, 1500);

 

} catch (err) {

// 4. 複製失敗的處理方式：顯示要複製的文字，引導手動複製

messageBox.innerHTML = `

<p class="font-bold text-xl" style="color: #D34343;">⚠️ 自動複製失敗！</p>

<p class="text-gray-700 mt-2">請**手動複製**下方預約資訊，然後點擊 LINE 連結聯繫我們：</p>

<textarea class="w-full mt-3 p-2 text-sm rounded border" rows="5" readonly>${bookingSummary}</textarea>

<a href="https://lin.ee/6w27dyj" target="_blank" class="w-full py-3 rounded-xl font-semibold text-white transition duration-300 hover:opacity-90 mt-4 block text-center" style="background-color: #06C755;">

<i class="fa-brands fa-line mr-2"></i> 點此前往 LINE (ID: @931qwaxz)

</a>

`;

messageBox.classList.remove('hidden');

messageBox.scrollIntoView({ behavior: 'smooth', block: 'start' });

}

}
